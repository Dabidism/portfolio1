DB Fiddle Link

RFID Table

CREATE TABLE RFIDTag (
stickerID VARCHAR(10) PRIMARY KEY,
tagCode VARCHAR(50),
issuedAt DATETIME,
status ENUM('active', 'inactive'),
expirationDate DATE);


TemporaryVehiclePass Table

CREATE TABLE TemporaryVehiclePass ( passID VARCHAR(10) PRIMARY KEY,
visitorID VARCHAR(10),
issueDate DATE,
expiryDate DATE,
status ENUM('active', 'expired') );


visitor Table

CREATE TABLE visitor ( visitorID VARCHAR(10) PRIMARY KEY,
passID VARCHAR(10),
fullName VARCHAR(100),
plateNum VARCHAR(20),
purposeOfVisit TEXT,
FOREIGN KEY (passID) REFERENCES TemporaryVehiclePass(passID) );


Vehicle Table

CREATE TABLE Vehicle ( plateNum VARCHAR(20) PRIMARY KEY,
OwnerID VARCHAR(10),
stickerID VARCHAR(10),
visitorID VARCHAR(10),
vehicleType VARCHAR(50),
model VARCHAR(50),
manufacturer VARCHAR(50),
color VARCHAR(30),
cubicCapacity INT,
numOfWheels INT, fuelType VARCHAR(30),
driverLicenseNum VARCHAR(30),
FOREIGN KEY (stickerID) REFERENCES RFIDTag(stickerID),
FOREIGN KEY (visitorID) REFERENCES visitor(visitorID) );


USER Table

CREATE TABLE USER ( userID VARCHAR(10) PRIMARY KEY,
registrationID VARCHAR(10),
passID VARCHAR(10),
username VARCHAR(50) NOT NULL UNIQUE,
password VARCHAR(255) NOT NULL,
role ENUM('student', 'employee', 'admin') NOT NULL );


document Table

CREATE TABLE document ( documentID VARCHAR(10) PRIMARY KEY,
registrationID VARCHAR(10),
documentType VARCHAR(50),
filePath VARCHAR(255),
uploadedAt DATETIME );


registration Table

CREATE TABLE registration ( registrationID VARCHAR(10) PRIMARY KEY,
userID VARCHAR(10),
plateNum VARCHAR(20),
documentID VARCHAR(10),
approvalCode VARCHAR(50),
registrationDate DATETIME,
FOREIGN KEY (userID) REFERENCES USER(userID),
FOREIGN KEY (plateNum) REFERENCES Vehicle(plateNum),
FOREIGN KEY (documentID) REFERENCES document(documentID) );


VehicleOwner Table

CREATE TABLE VehicleOwner ( OwnerID VARCHAR(10) PRIMARY KEY,
plateNum VARCHAR(20),
registrationID VARCHAR(10),
fName VARCHAR(50),
lName VARCHAR(50),
role VARCHAR(30),
department VARCHAR(50),
email VARCHAR(100),
course VARCHAR(50),
year VARCHAR(10),
section VARCHAR(10),
academicYear VARCHAR(20),
designation VARCHAR(50),
employmentStatus VARCHAR(50),
registrationStatus ENUM('pending', 'approved', 'rejected'),
FOREIGN KEY (plateNum) REFERENCES Vehicle(plateNum),
FOREIGN KEY (registrationID) REFERENCES registration(registrationID) );


EntryExitLog Table

CREATE TABLE EntryExitLog ( logID VARCHAR(10) PRIMARY KEY,
plateNum VARCHAR(20),
entryTime DATETIME,
exitTime DATETIME,
rfid BOOLEAN,
status ENUM('entered', 'exited', 'denied'),
gateLocation VARCHAR(100),
FOREIGN KEY (plateNum) REFERENCES Vehicle(plateNum) );


Visitor Log Table

CREATE TABLE VisitorLog ( visitorLogID VARCHAR(10) PRIMARY KEY,
visitorID VARCHAR(10),
plateNum VARCHAR(20),
entryTime DATETIME,
exitTime DATETIME,
status ENUM('entered', 'exited', 'denied'),
FOREIGN KEY (visitorID) REFERENCES visitor(visitorID),
FOREIGN KEY (plateNum) REFERENCES Vehicle(plateNum) );


ANPRLog Table

CREATE TABLE ANPRLog ( ANPRLogID VARCHAR(10) PRIMARY KEY,
plateNum VARCHAR(20),
status ENUM('matched', 'unmatched'),
location VARCHAR(100),
FOREIGN KEY (plateNum) REFERENCES Vehicle(plateNum) );
